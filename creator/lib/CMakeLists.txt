    plugin.cpp
    dockplugin.cpp
    pluginmanager.cpp

    add_dependencies(libGluonCreator GluonEngine)
    target_link_libraries(libGluonCreator GluonEngine ${KDE4_KDEUI_LIBS})

    target_link_libraries(libGluonCreator ${GLUON_ENGINE_LIBS} ${KDE4_KDEUI_LIBS})
endif()
